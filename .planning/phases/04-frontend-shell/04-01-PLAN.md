From the project root (C:\Users\Nowaxial\Documents\Projekt\Biine\), run:

```
npm create astro@latest Biine.WEB -- --template minimal --typescript strict --no-git --install
```

Then inside Biine.WEB/, install dependencies:

```
cd Biine.WEB
npm install @astrojs/react @mantine/core @mantine/hooks @vite-pwa/astro react react-dom
npm install -D @types/react @types/react-dom
```

Then update `astro.config.mjs` to:

```js
import { defineConfig } from 'astro/config';
import react from '@astrojs/react';
import { VitePWA } from '@vite-pwa/astro';

export default defineConfig({
  integrations: [
    react(),
    VitePWA({
      registerType: 'autoUpdate',
      manifest: {
        name: 'Biine',
        short_name: 'Biine',
        description: 'Where Bite meets Dine',
        theme_color: '#e8590c',
        background_color: '#ffffff',
        display: 'standalone',
        start_url: '/',
        icons: [
          { src: '/icons/icon-192.png', sizes: '192x192', type: 'image/png' },
          { src: '/icons/icon-512.png', sizes: '512x512', type: 'image/png' },
        ],
      },
      workbox: {
        globPatterns: ['**/*.{js,css,html,ico,png,svg}'],
      },
    }),
  ],
  vite: {
    define: {},
  },
});
```

Create `Biine.WEB/.env`:
```
VITE_API_URL=http://localhost:5035
```

Update `tsconfig.json` to include `"jsx": "react-jsx"` in compilerOptions if not already set by the template.

**MantineProvider Island:** Create `Biine.WEB/src/components/MantineShell.tsx`:

```tsx
import { MantineProvider } from '@mantine/core';
import '@mantine/core/styles.css';
import type { ReactNode } from 'react';

interface Props {
  children: ReactNode;
}

export default function MantineShell({ children }: Props) {
  return (
    <MantineProvider defaultColorScheme="light">
      {children}
    </MantineProvider>
  );
}
```

**Index page:** Create `Biine.WEB/src/pages/index.astro`:

```astro
---
import BaseLayout from '../layouts/BaseLayout.astro';
import MantineShell from '../components/MantineShell';
import { Text, Stack, Center } from '@mantine/core';
---
<BaseLayout title="Biine">
  <MantineShell client:load>
    <Center style={{ height: '100dvh' }}>
      <Stack align="center" gap="md">
        <Text size="xl" fw={700} c="orange.7">Biine 🍽️</Text>
        <Text c="dimmed">Where Bite meets Dine</Text>
      </Stack>
    </Center>
  </MantineShell>
</BaseLayout>
```

Note: Mantine v7+ imports `@mantine/core/styles.css` inside the React component or a global CSS file. Verify this import path works after install; if it doesn't, try `@mantine/core/styles/global.css`.
