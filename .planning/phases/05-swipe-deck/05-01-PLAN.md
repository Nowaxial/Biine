**Create `Biine.WEB/src/hooks/useSession.ts`:**

Generates a UUID session ID on first call, persists in localStorage as `biine_session_id`,
returns the same ID on subsequent calls.

```ts
import { useState, useEffect } from 'react';

const SESSION_KEY = 'biine_session_id';

function generateId(): string {
  if (typeof crypto !== 'undefined' && crypto.randomUUID) {
    return crypto.randomUUID();
  }
  // Fallback for older environments
  return Math.random().toString(36).slice(2) + Date.now().toString(36);
}

export function useSession(): string {
  const [sessionId, setSessionId] = useState<string>('');

  useEffect(() => {
    let id = localStorage.getItem(SESSION_KEY);
    if (!id) {
      id = generateId();
      localStorage.setItem(SESSION_KEY, id);
    }
    setSessionId(id);
  }, []);

  return sessionId;
}
```

**Create `Biine.WEB/src/hooks/useRecipe.ts`:**

Fetches the next unseen recipe for the session. Manages loading/recipe/empty/error states.
Exposes a `fetchNext()` function to load the next recipe after a swipe.

```ts
import { useState, useCallback } from 'react';

const API_URL = import.meta.env.VITE_API_URL as string;

export interface Recipe {
  id: number;
  nameSv: string;
  nameEn: string;
  personaTextSv: string;
  personaTextEn: string;
  cookingTime: number;
  difficulty: string;
  cuisine: string;
  tags: string[];
  imageUrl: string;
}

type State =
  | { status: 'idle' }
  | { status: 'loading' }
  | { status: 'recipe'; data: Recipe }
  | { status: 'empty' }
  | { status: 'error'; message: string };

export function useRecipe(sessionId: string) {
  const [state, setState] = useState<State>({ status: 'idle' });

  const fetchNext = useCallback(async () => {
    if (!sessionId) return;
    setState({ status: 'loading' });
    try {
      const res = await fetch(
        `${API_URL}/api/recipes/next?sessionId=${sessionId}&lang=sv`
      );
      if (res.status === 204) {
        setState({ status: 'empty' });
        return;
      }
      if (res.status === 429) {
        setState({ status: 'error', message: 'För många förfrågningar. Vänta lite.' });
        return;
      }
      if (!res.ok) {
        setState({ status: 'error', message: 'Kunde inte hämta recept.' });
        return;
      }
      const data: Recipe = await res.json();
      setState({ status: 'recipe', data });
    } catch {
      setState({ status: 'error', message: 'Nätverksfel. Kontrollera anslutningen.' });
    }
  }, [sessionId]);

  return { state, fetchNext };
}
```

**Create `Biine.WEB/src/components/SwipeDeck.tsx`:**

Fetches the first recipe on mount. Uses react-tinder-card for swipe gestures.
On swipe right → "like", left → "pass". After swipe, POSTs interaction then fetches next recipe.

```tsx
import { useEffect, useRef } from 'react';
import TinderCard from 'react-tinder-card';
import { Center, Stack, Text, Loader, Button } from '@mantine/core';
import { useSession } from '../hooks/useSession';
import { useRecipe } from '../hooks/useRecipe';
import RecipeCard from './RecipeCard';

const API_URL = import.meta.env.VITE_API_URL as string;

export default function SwipeDeck() {
  const sessionId = useSession();
  const { state, fetchNext } = useRecipe(sessionId);
  const swipedRef = useRef(false);

  // Fetch first recipe once session ID is ready
  useEffect(() => {
    if (sessionId && state.status === 'idle') {
      fetchNext();
    }
  }, [sessionId]);

  const handleSwipe = async (direction: string) => {
    if (swipedRef.current) return;
    if (state.status !== 'recipe') return;
    swipedRef.current = true;

    const action = direction === 'right' ? 'like' : 'pass';

    // Fire and forget — POST interaction
    fetch(`${API_URL}/api/interactions`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ sessionId, recipeId: state.data.id, action }),
    }).catch(() => {/* ignore network errors on interaction logging */});

    // Fetch next recipe
    await fetchNext();
    swipedRef.current = false;
  };

  if (state.status === 'idle' || state.status === 'loading') {
    return (
      <Center style={{ height: '100dvh' }}>
        <Loader color="orange" size="lg" />
      </Center>
    );
  }

  if (state.status === 'empty') {
    return (
      <Center style={{ height: '100dvh' }}>
        <Stack align="center" gap="md">
          <Text size="64px">🍽️</Text>
          <Text size="xl" fw={700} ta="center">Du har sett allt!</Text>
          <Text c="dimmed" ta="center">Inga fler recept just nu.</Text>
        </Stack>
      </Center>
    );
  }

  if (state.status === 'error') {
    return (
      <Center style={{ height: '100dvh' }}>
        <Stack align="center" gap="md">
          <Text size="xl" fw={700} c="red" ta="center">Något gick fel</Text>
          <Text c="dimmed" ta="center">{state.message}</Text>
          <Button color="orange" onClick={fetchNext}>Försök igen</Button>
        </Stack>
      </Center>
    );
  }

  // state.status === 'recipe'
  return (
    <Center style={{ height: '100dvh', padding: '16px' }}>
      <div style={{ position: 'relative', width: '100%', maxWidth: 380, height: 520 }}>
        <TinderCard
          key={state.data.id}
          onSwipe={handleSwipe}
          preventSwipe={['up', 'down']}
          swipeRequirementType="position"
          swipeThreshold={80}
        >
          <RecipeCard recipe={state.data} />
        </TinderCard>
      </div>
    </Center>
  );
}
```
